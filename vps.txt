set up a vps for simple hosting

# add user
$ adduser tga

# add sudo priviliedge to user
$ usermod -aG sudo tga

# change to user
$ su - tga

# install packages
$ sudo apt update
$ sudo apt install -y vim make git fish unzip sqlite3

# install bun
$ curl -fsSL https://bun.sh/install | bash

# install caddy
$ sudo apt install -y debian-keyring debian-archive-keyring apt-transport-https
$ curl -1sLf 'https://dl.cloudsmith.io/public/caddy/stable/gpg.key' | sudo gpg --dearmor -o /usr/share/keyrings/caddy-stable-archive-keyring.gpg
$ curl -1sLf 'https://dl.cloudsmith.io/public/caddy/stable/debian.deb.txt' | sudo tee /etc/apt/sources.list.d/caddy-stable.list
$ sudo apt update
$ sudo apt install -y caddy

# clone this repo
$ git clone https://github.com/slmjkdbtl/rc

# init config
$ cd rc && make

# add public ssh key
$ vim .ssh/authorized_keys

# change default shell
$ chsh -s /usr/bin/fish

# reload systemd configs files
$ systemctl daemon-reload

# start a service
$ systemctl start space55.xyz

# stop a service
$ systemctl stop space55.xyz

# enable a service to run on startup
$ systemctl enable space55.xyz

# reload config for a service
$ systemctl reload caddy

# view log
journalctl -qu space55.xyz
```

# /etc/systemd/system/space55.xyz.service
[Unit]
Description=space55.xyz
After=network.target

[Service]
Type=simple
User=tga
Environment="PORT=4000"
WorkingDirectory=/home/tga/space55.xyz
StandardError=append:/home/tga/space55.xyz/data/err.log
ExecStart=/home/tga/.bun/bin/bun main.js
Restart=always
StandardOutput=journal
StandardError=journal
StandardInput=null

[Install]
WantedBy=multi-user.target

# /etc/caddy/Caddyfile
space55.xyz {
        reverse_proxy localhost:4000
}
